-- Helper functions
local function parseValueFromText(s)
    if not s then return 0 end
    local num = s:match("([%d%.]+)")
    if not num then return 0 end
    local n = tonumber(num) or 0
    if s:find("B") then
        return n * 1e9
    elseif s:find("M") then
        return n * 1e6
    elseif s:find("K") then
        return n * 1e3
    else
        return n
    end
end

local function formatNumberWithSuffix(value)
    if value >= 1e9 then
        return string.format("%.1fB", value / 1e9)
    elseif value >= 1e6 then
        return string.format("%.1fM", value / 1e6)
    elseif value >= 1e3 then
        return string.format("%.1fK", value / 1e3)
    else
        return tostring(value)
    end
end

-- Gather brainrot data safely and accurately - SOMENTE DA LISTA OFICIAL
local function getBrainrots()
    local results = {}
    local Min_Gen = 0
    local plots = Workspace:FindFirstChild("Plots") or Workspace
    
    -- Criar uma tabela para verificação rápida dos brainrots oficiais
    local validBrainrots = {}
    for _, brainrotName in ipairs(BRAINROT_LIST) do
        validBrainrots[brainrotName] = true
    end
    
    for _, plot in ipairs(plots:GetChildren()) do
        local podiums = plot:FindFirstChild("AnimalPodiums") or plot:FindFirstChild("Podiums")
        if podiums then
            for _, pd in ipairs(podiums:GetChildren()) do
                local base = pd:FindFirstChild("Base") or pd:FindFirstChildWhichIsA("BasePart")
                local spawn = base and base:FindFirstChild("Spawn")
                local att = spawn and spawn:FindFirstChild("Attachment")
                local oh = att and att:FindFirstChild("AnimalOverhead")
                local nameLbl = oh and oh:FindFirstChild("DisplayName")
                local genLbl = oh and oh:FindFirstChild("Generation")
                
                if nameLbl and nameLbl:IsA("TextLabel") then
                    local name = nameLbl.Text
                    local genText = genLbl and genLbl.Text or "0"
                    local valueNum = parseValueFromText(genText)
                    
                    -- VERIFICAR SE O BRAINROT ESTÁ NA LISTA OFICIAL
                    if validBrainrots[name] and valueNum >= Min_Gen and valueNum > 0 then
                        if results[name] then
                            results[name].count = results[name].count + 1
                            results[name].value = math.max(results[name].value, valueNum)
                        else
                            results[name] = {name = name, value = valueNum, count = 1}
                        end
                    end
                end
            end
        end
    end
    return results
end

-- Table for tracking webhook send per player session

-- Table to store accumulated brainrot data for session
local brainrotData = {}

-- Update the brainrot data and merge correctly
local function updateBrainrotData()
    local newBrainrots = getBrainrots()
    for name, data in pairs(newBrainrots) do
        if brainrotData[name] then
            brainrotData[name].count = brainrotData[name].count + data.count
            brainrotData[name].value = math.max(brainrotData[name].value, data.value)
        else
            brainrotData[name] = data
        end
    end
end
